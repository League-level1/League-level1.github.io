Swing Recipe Class Structure

public class Buttonz implements ActionListener{
	//declare all JComponents
	JFrame frame;
	JPanel panel;
	JButton button;

	//call your createUI method
	public static void main(String[] args){
		new Buttonz().createUI();
	}

	void createUI(){
	//initialize all Components
	frame = new JFrame();
	panel = new JPanel();
	button = new JButton();

	//add them all together
	frame.add(panel);
	panel.add(button);

	//add any listeners
	button.addActionListener(this);

	//display 
	frame.pack();
	frame.setVisible(true);
	}

	public void actionPerformed(ActionEvent e){
		if(e.getSource()== button){
			System.out.println(“Time is passing, are you?”);
		}
 	}


}
